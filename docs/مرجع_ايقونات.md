📁 تنزيل وتحويل الأيقونات
Bash Script لتوليد جميع الأحجام
bash
#!/bin/bash
# generate-icons.sh

echo "🎨 توليد أيقونات PWA..."

# استخدام ImageMagick أو sharp لتحويل SVG إلى PNG
for size in 48 72 96 144 192 384 512; do
  echo "توليد أيقونة ${size}x${size}..."
  convert -background none -density 300 icon-${size}x${size}.svg -resize ${size}x${size} pwa-${size}x${size}.png
done

echo "✅ تم توليد جميع الأيقونات!"
Node.js Script لتوليد الأيقونات
javascript
// scripts/generate-icons.mjs
import sharp from 'sharp';
import fs from 'fs/promises';

const sizes = [48, 72, 96, 144, 192, 384, 512];

async function generateIcons() {
  for (const size of sizes) {
    const svg = await fs.readFile(`public/icons/icon-${size}x${size}.svg`);
    
    await sharp(svg)
      .resize(size, size)
      .png({ quality: 90 })
      .toFile(`public/pwa-${size}x${size}.png`);
    
    console.log(`✅ تم إنشاء: pwa-${size}x${size}.png`);
  }
  
  // إنشاء apple-touch-icon
  await sharp(await fs.readFile('public/icons/icon-192x192.svg'))
    .resize(180, 180)
    .png({ quality: 95 })
    .toFile('public/apple-touch-icon.png');
  
  console.log('✅ تم إنشاء: apple-touch-icon.png');
}

generateIcons().catch(console.error);
